import dedent                   from 'ts-dedent'
import CodeBlock                from '@theme/CodeBlock'
import { CUSTOM_VALUE }         from '@site/src/constants/kubernetes/customValue'


> Данная команда запускает сервис Kubelet в рамках выделенной фазы утилиты Kubeadm.


<h4>Запуск kubelet</h4>
<CodeBlock language="bash">
  {dedent`
    kubeadm init phase kubelet-start \\
      --config=${CUSTOM_VALUE.kubeadmBaseConfigPath.value}/kubeadm.yaml \\
      --kubeconfig=${CUSTOM_VALUE.kuberneteBaseFolderPath.value}/super-admin.conf
  `}
</CodeBlock>

:::note
<CodeBlock language="bash">
  {dedent`
    [kubelet-start] Writing kubelet environment file with flags to file "/var/lib/kubelet/kubeadm-flags.env"
    [kubelet-start] Writing kubelet configuration to file "/var/lib/kubelet/config.yaml"
    [kubelet-start] Starting the kubelet
  `}
</CodeBlock>
:::
