import dedent                   from 'ts-dedent'
import CodeBlock                from '@theme/CodeBlock'

import { KUBEADM_DOWNLOAD_SCRIPT }     from '@site/i18n/en/docusaurus-plugin-content-docs/current/tech-docs/kubernetes/components/kubeadm/download-script.sh'

import { DOWNLOAD_VERSION }     from '@site/src/constants/kubernetes/downloads'
import { COMPONENTS_VERSION }   from '@site/src/constants/kubernetes/componentsVersion'
import TabItem from '@theme/TabItem'
import Tabs from '@theme/Tabs'

<Tabs groupId="install-type">

    <TabItem value='Bash'>
        <h4>Creating working directories</h4>
        <CodeBlock language="bash">
            {dedent`
              mkdir -p /etc/default/kubeadm
            `}
        </CodeBlock>

        <h4>Environment variables</h4>
        <CodeBlock language="bash">
            {dedent`
              cat <<EOF > /etc/default/kubeadm/download.env
              COMPONENT_VERSION="${COMPONENTS_VERSION.kubernetes.value}"
              REPOSITORY="${DOWNLOAD_VERSION.kubeadm.baseUrl}"
              EOF
            `}
        </CodeBlock>

        <h4>Download instructions</h4>
        <CodeBlock language="bash">
          {dedent`
            cat <<"EOF" > /etc/default/kubeadm/download-script.sh
            ${KUBEADM_DOWNLOAD_SCRIPT.data.value}
            EOF
          `}
        </CodeBlock>

        <h4>Permission setup</h4>
        <CodeBlock language="bash">
          {dedent`
            chmod +x /etc/default/kubeadm/download-script.sh
          `}
        </CodeBlock>

        <h4>Download service</h4>
        <CodeBlock language="bash">
          {dedent`
            cat <<EOF > /usr/lib/systemd/system/kubeadm-install.service
            [Unit]
            Description=Install and update kubeadm
            After=network.target
            Wants=network-online.target

            [Service]
            Type=oneshot
            EnvironmentFile=-/etc/default/kubeadm/download.env
            ExecStart=/bin/bash -c "/etc/default/kubeadm/download-script.sh"
            RemainAfterExit=yes

            [Install]
            WantedBy=multi-user.target
            EOF
          `}
        </CodeBlock>

        <h4>Download</h4>
        <CodeBlock language="bash">
          {dedent`
            systemctl enable kubeadm-install.service
            systemctl start kubeadm-install.service
          `}
        </CodeBlock>

    </TabItem>

    <TabItem value='Cloud-init'>

        <h4>Environment variables</h4>
        <CodeBlock language="yaml">
          {dedent`
            - path: /etc/default/kubeadm/download.env
              owner: root:root
              permissions: '0644'
              content: |
                COMPONENT_VERSION="${COMPONENTS_VERSION.kubernetes.value}"

        <h4>Download instructions/Permission setup</h4>
        <CodeBlock language="yaml">

            - path: /etc/default/kubeadm/download-script.sh
              owner: root:root
              permissions: '0755'
              content: |
                ${KUBEADM_DOWNLOAD_SCRIPT.data.value}
          `}
        </CodeBlock>

        <h4>Download service</h4>
        <CodeBlock language="yaml">
          {dedent`
            - path: /usr/lib/systemd/system/kubeadm-install.service
              owner: root:root
              permissions: '0644'
              content: |
                [Unit]
                Description=Install and update kubeadm
                After=network.target
                Wants=network-online.target

                [Service]
                Type=oneshot
                EnvironmentFile=-/etc/default/kubeadm/download.env
                ExecStart=/bin/bash -c "/etc/default/kubeadm/download-script.sh"
                RemainAfterExit=yes

                [Install]
                WantedBy=multi-user.target
          `}
        </CodeBlock>

        <h4>Download</h4>
        <CodeBlock language="bash">
          {dedent`
            - systemctl enable kubeadm-install.service
            - systemctl start kubeadm-install.service
          `}
        </CodeBlock>

    </TabItem>

</Tabs>
