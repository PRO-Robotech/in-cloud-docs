import dedent from 'ts-dedent'
import CodeBlock from '@theme/CodeBlock'
import { CUSTOM_VALUE } from '@site/src/constants/kubernetes/customValue'

:::warning Note
Note that when creating a cluster with **_Kubeadm_** without running `kubeadm init` or `kubeadm join`, the Kubelet configuration file (`/var/lib/kubelet/config.yaml`) will not be created.
:::

<CodeBlock language="bash">
  {dedent`
        ls -la ${CUSTOM_VALUE.kuberneteKubeletFolderPath.value}/config-custom.yaml
    `}
</CodeBlock>

<CodeBlock language="bash">
  {dedent`
        ls -la /usr/lib/systemd/system/kubelet.service.d/10-kubeadm.conf
    `}
</CodeBlock>

:::note Command output

<CodeBlock language="bash">
  {dedent`
        -rw-r--r-- 1 root root 1721 Feb 19 18:57 /var/lib/kubelet/config.yaml
    `}
</CodeBlock>

<CodeBlock language="bash">
  {dedent`
        -rw-r--r-- 1 root root 903 Feb 19 22:10 /usr/lib/systemd/system/kubelet.service.d/10-kubeadm.conf
    `}
</CodeBlock>
:::
