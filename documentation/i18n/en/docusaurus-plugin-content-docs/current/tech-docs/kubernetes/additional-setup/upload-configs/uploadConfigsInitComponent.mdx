import TabItem from '@theme/TabItem'
import Tabs from '@theme/Tabs'

import CodeBlock from '@theme/CodeBlock'
import dedent from 'ts-dedent'

import { KUBEADM_CONFIG_DATA } from '@site/src/constants/kubernetes/kubeadmConfigData'
import { KUBELET_CONFIG_DATA } from '@site/src/constants/kubernetes/kubeletConfigData'
import { CUSTOM_VALUE } from '@site/src/constants/kubernetes/customValue'
import { ETCD_ARGS } from '@site/src/constants/kubernetes/etcdArgs'

:::warning Note
This section describes the instructions for uploading the current `Kubeadm` and `Kubelet` configuration to the Kubernetes control plane as a `ConfigMap` resource. This approach simplifies managing configuration changes for Kubernetes nodes, covering both `worker` and `master` nodes.
:::

<Tabs groupId="install-type">

  <TabItem value='HardWay'>
    <h4>Environment variables for configuration file template</h4>
    <CodeBlock language="bash">
      {dedent`
        export CLUSTER_NAME='${CUSTOM_VALUE.clusterName.value}'
        export BASE_DOMAIN='${CUSTOM_VALUE.kubernetesBaseDomain.value}'
        export FULL_HOST_NAME=${CUSTOM_VALUE.virtualMachineFullName.value}
        export INTERNAL_API=api.${CUSTOM_VALUE.kubernetesClusterExternalDomain.value}
        export MACHINE_LOCAL_ADDRESS=${CUSTOM_VALUE.virtualMachineLocalAddress.value}
        export ETCD_INITIAL_CLUSTER="$\{FULL_HOST_NAME}=${ETCD_ARGS.initialAdvertisePeerUrls.value}"
        export AUTH_EXTRA_GROUPS="system:bootstrappers:kubeadm:default-node-token"
      `}
    </CodeBlock>

    <details className="custom-gray-block">
      <summary>kubeadm-config</summary>

      > This block is required to allow nodes to read the `kubeadm-config` ConfigMap in the `kube-system` namespace:

      <CodeBlock>
        {dedent`
          kubectl \\
            --kubeconfig=${CUSTOM_VALUE.kubernetesBaseFolderPath.value}/super-admin.conf \\
            apply -f - <<EOF
          ---
          apiVersion: rbac.authorization.k8s.io/v1
          kind: Role
          metadata:
            name: kubeadm:nodes-kubeadm-config
            namespace: kube-system
          rules:
          - apiGroups:
            - ""
            resourceNames:
            - kubeadm-config
            resources:
            - configmaps
            verbs:
            - get
          ---
          apiVersion: rbac.authorization.k8s.io/v1
          kind: RoleBinding
          metadata:
            name: kubeadm:nodes-kubeadm-config
            namespace: kube-system
          roleRef:
            apiGroup: rbac.authorization.k8s.io
            kind: Role
            name: kubeadm:nodes-kubeadm-config
          subjects:
          - apiGroup: rbac.authorization.k8s.io
            kind: Group
            name: $\{AUTH_EXTRA_GROUPS}
          - apiGroup: rbac.authorization.k8s.io
            kind: Group
            name: system:nodes
          EOF
        `}
      </CodeBlock>

      > This block is required so that when executing `kubeadm join`, the node receives the current `ClusterConfiguration` from the control cluster and correctly joins the control-plane.

      <CodeBlock>
        {dedent`
          kubectl \\
            --kubeconfig=${CUSTOM_VALUE.kubernetesBaseFolderPath.value}/super-admin.conf \\
            apply -f - <<EOF
          ---
          apiVersion: v1
          kind: ConfigMap
          metadata:
            name: kubeadm-config
            namespace: kube-system
          data:
            ClusterConfiguration: |
              ${KUBEADM_CONFIG_DATA.data.value}
          EOF
        `}

      </CodeBlock>
    </details>

    <details className="custom-gray-block">
      <summary>kubelet-config</summary>

      > This block is required to allow nodes to read the `kubelet-config` ConfigMap in the `kube-system` namespace:

      <CodeBlock>

          apiVersion: rbac.authorization.k8s.io/v1

          metadata:

            namespace: kube-system
          rules:
          - apiGroups:
            - ""
            resourceNames:
            - kubelet-config
            resources:
            - configmaps
            verbs:
            - get
          ---
          apiVersion: rbac.authorization.k8s.io/v1
          kind: RoleBinding
          metadata:
            name: kubeadm:kubelet-config
            namespace: kube-system
          roleRef:
            apiGroup: rbac.authorization.k8s.io
            kind: Role

          subjects:
          - apiGroup: rbac.authorization.k8s.io
            kind: Group
            name: system:nodes
          - apiGroup: rbac.authorization.k8s.io
            kind: Group
            name: $\{AUTH_EXTRA_GROUPS}
          EOF
        `}
      </CodeBlock>

      > This block is required so that when executing `kubeadm join`, the node receives the current `kubelet-config` from the control cluster and correctly joins the control-plane.

      <CodeBlock>

        `}
      </CodeBlock>
    </details>

  </TabItem>

  <TabItem value='Kubeadm'>

    <h4>Configuration update</h4>
    <CodeBlock language="bash">
      {dedent`
        kubeadm init phase upload-config all \\
          --config=${CUSTOM_VALUE.kubeadmBaseConfigPath.value}/kubeadm.yaml \\
          --kubeconfig=${CUSTOM_VALUE.kubernetesBaseFolderPath.value}/super-admin.conf
      `}
    </CodeBlock>

    :::note
    <CodeBlock language="bash">
      {dedent`
        [upload-config] Storing the configuration used in ConfigMap "kubeadm-config" in the "kube-system" Namespace
        [kubelet] Creating a ConfigMap "kubelet-config" in namespace kube-system with the configuration for the kubelets in the cluster
      `}
    </CodeBlock>
    :::

  </TabItem>
</Tabs>
